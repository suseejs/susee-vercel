git submodule add --name susee https://github.com/phothinmg/susee.git susee

cd .git/modules/susee
git config core.repositoryformatversion 1
git sparse-checkout init --cone
git sparse-checkout set coverage

git submodule update --init --recursive --depth 1 susee

# 1. Unregister the submodule from local configurations
git submodule deinit -f path/to/submodule

# 2. Remove the submodule directory from the working tree and Git index
git rm -f path/to/submodule

# 3. Clean up internal Git cached directory metadata
rm -rf .git/modules/path/to/submodule

# 4. Commit the change to finalize the removal
git commit -m "Remove submodule"
